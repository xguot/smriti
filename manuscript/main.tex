\section{Algorithm}

Given an initial imputation $\tilde{\mathbf{X}} \in \mathbb{R}^{n \times p}$, a missingness mask $\mathbf{M} \in \{0,1\}^{n \times p}$, and a target covariance $\boldsymbol{\Sigma}^* \in \mathbb{R}^{p \times p}$, minimize:

\begin{equation}
\mathcal{L}(\hat{\mathbf{X}}) = \|\hat{\mathbf{X}} - \tilde{\mathbf{X}}\|_F^2 \;+\; \lambda \cdot \|\operatorname{cov}(\hat{\mathbf{X}}) - \boldsymbol{\Sigma}^*\|_F^2
\end{equation}

subject to $\hat{X}_{ij} = \tilde{X}_{ij}$ wherever $M_{ij} = 0$ (observed values frozen).

\textbf{Gradient.} Let $\bar{\mathbf{X}} = \hat{\mathbf{X}} - \mathbf{1}_n\hat{\boldsymbol{\mu}}^\top$, $\hat{\boldsymbol{\Sigma}} = \frac{1}{n-1}\bar{\mathbf{X}}^\top\bar{\mathbf{X}}$. Then:

\begin{equation}
\frac{\partial}{\partial \hat{\mathbf{X}}} \|\hat{\boldsymbol{\Sigma}} - \boldsymbol{\Sigma}^*\|_F^2 = \frac{4}{n-1} \cdot \bar{\mathbf{X}} \, (\hat{\boldsymbol{\Sigma}} - \boldsymbol{\Sigma}^*)
\end{equation}

\textbf{Update.} $\hat{\mathbf{X}} \leftarrow \hat{\mathbf{X}} - \eta \cdot \lambda \cdot 2\bar{\mathbf{X}}(\hat{\boldsymbol{\Sigma}} - \boldsymbol{\Sigma}^*) \odot \mathbf{M}$

\begin{algorithm}[ht]
\caption{Lagrangian Projection (75 lines C++)}
\begin{algorithmic}[1]
\State \textbf{Input:} $\tilde{\mathbf{X}}$, $\mathbf{M}$, $\boldsymbol{\Sigma}^*$, $\lambda$, $\eta$, $\epsilon$
\State $\hat{\mathbf{X}} \gets \tilde{\mathbf{X}}$
\State Ensure $\boldsymbol{\Sigma}^* \succeq 0$ (zero negative eigenvalues)
\For{$t = 1$ to $\text{max\_iter}$}
    \State $\bar{\mathbf{X}} \gets \hat{\mathbf{X}} - \mathbf{1}_n\hat{\boldsymbol{\mu}}^\top$
    \State $\hat{\boldsymbol{\Sigma}} \gets \frac{1}{n-1}\bar{\mathbf{X}}^\top\bar{\mathbf{X}}$
    \If{$\|\hat{\boldsymbol{\Sigma}} - \boldsymbol{\Sigma}^*\|_F < \epsilon$} \Return $\hat{\mathbf{X}}$ \EndIf
    \State $\mathbf{G} \gets 2 \cdot \bar{\mathbf{X}} \cdot (\hat{\boldsymbol{\Sigma}} - \boldsymbol{\Sigma}^*)$
    \State $\hat{\mathbf{X}} \gets \hat{\mathbf{X}} - (\eta \cdot \lambda) \cdot \mathbf{G} \odot \mathbf{M}$
\EndFor
\State \Return $\hat{\mathbf{X}}$
\end{algorithmic}
\end{algorithm}

\section{Pipeline}

\textbf{Step 1.} missForest $\to$ initial imputation $\tilde{\mathbf{X}}$.

\textbf{Step 2.} lavaan FIML $\to$ model-implied $\boldsymbol{\Sigma}^*_{\text{FIML}}$.

\textbf{Step 3.} Run Algorithm~1 with $\tilde{\mathbf{X}}$ and $\boldsymbol{\Sigma}^*_{\text{FIML}}$. Output $\hat{\mathbf{X}}$ is a completed dataset with $\operatorname{cov}(\hat{\mathbf{X}}) \approx \boldsymbol{\Sigma}^*_{\text{FIML}}$.
